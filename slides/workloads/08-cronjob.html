CronJob

  • A CronJob runs Jobs on a schedule.
  • nightly cleanup
  • scheduled reports
  • periodic sync
  • backups
  • set concurrencyPolicy
apiVersion: batch/v1
kind: CronJob
metadata:
  name: cleanup
spec:
  schedule: "0 2 * * *"
  concurrencyPolicy: Forbid
  successfulJobsHistoryLimit: 3
  failedJobsHistoryLimit: 3
  jobTemplate:
    spec:
      template:
        spec:
          restartPolicy: Never
          containers:
            - name: cleanup
              image: busybox
              command: ["sh", "-c", "echo cleanup"]
